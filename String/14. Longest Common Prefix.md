### string manipulation

```java
class Solution {
    public String longestCommonPrefix(String[] strs) {
        // find the shortest string
        int min = 200;
        String shortest = "";
        for(String str: strs) {
            if(str.length() < min) {
                min = str.length();
                shortest = str;
            }
        }
        int flag = 0;
        String prefix = "";
        for(int i = shortest.length(); i >= 0; i--) {
            flag = 0;
            prefix = shortest.substring(0, i);
            for(String str: strs) {
                if(!str.startsWith(prefix)) {
                    flag = 1;
                    break;
                }
            }
            if(flag == 0) break;
        }
        return prefix;
        
    }
}
```
