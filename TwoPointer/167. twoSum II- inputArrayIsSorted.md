### ==using 2 pointers==

```java

class Solution {
    public int[] twoSum(int[] numbers, int target) {
        //2 pointers
        int left = 0;
        int right = numbers.length -1;
        int[] res = new int[2];
        while(left < right){
            if(numbers[left] + numbers[right] == target){
                res[0] = left + 1;
                res[1] = right + 1;
                break;
            }
            else if(numbers[left] + numbers[right] > target){
                right--;
              	continue;
            }
            else{
                left++;
            }


        }
        return res;
        
    }
}
```
